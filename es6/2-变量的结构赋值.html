数组的解构赋值 
			1.如果解构不成功，变量的值就等于undefined。
			let [ , , third] = ["foo", "bar", "baz"];
			third // "baz"
			
			let [x, , y] = [1, 2, 3];
			x // 1
			y // 3
			
			let [head, ...tail] = [1, 2, 3, 4];
			head // 1
			tail // [2, 3, 4]
			
			let [x, y, ...z] = ['a'];
			x // "a"
			y // undefined
			z // []


			2.如果等号的右边不是数组（或者严格地说，不是可遍历的结构，参见《Iterator》一章），那么将会报错	
			// 报错
			let [foo] = 1;
			let [foo] = false;
			let [foo] = NaN;
			let [foo] = undefined;
			let [foo] = null;
			let [foo] = {};
			
			
			
			
			3.默认值  解构赋值允许指定默认值。
			et [foo = true] = [];
			foo // true
			
			let [x, y = 'b'] = ['a']; // x='a', y='b'
			let [x, y = 'b'] = ['a', undefined]; // x='a', y='b'
			
			4.当一个数组成员严格等于undefined，默认值才会生效。
			let [x = 1] = [undefined];
			x // 1
			
			let [x = 1] = [null];
			x // null
2.对象的解构赋值 
		  1.而对象的属性没有次序，变量必须与属性同名，才能取到正确的值。
			
			let { bar, foo } = { foo: 'aaa', bar: 'bbb' };
			foo // "aaa"
			bar // "bbb"
			
			let { baz } = { foo: 'aaa', bar: 'bbb' };
			baz // undefined
