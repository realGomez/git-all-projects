function add(a,b){
				//arguments 存实参	
				a=2;
				console.log(arguments[0]); //2
			}
			
			function add(a,b){
				//arguments 存实参				
				arguments[0] = 3;
				console.log(a);//3
			}
			
			add(1,2);
